def so_agent(event: Dict[str, Any] = None, doc_name: str = None) -> str:
    # get doc_name from event or direct call
    if event:
        doc_name = event.get("doc_name")
    if not doc_name:
        return "Missing required parameter: doc_name"

    transport = partial(streamablehttp_client, FAC_MCP_URL, headers=FAC_MCP_HEADERS)

    with MCPClient(transport) as mcp_client:
        tools = mcp_client.list_tools_sync()

        so_agent_instance = Agent(
            model=anthropic_model,
            system_prompt=SO_SYSTEM_PROMPT,
            tools=tools,
        )

        response = so_agent_instance(
            f'Use get_document with doctype="Sales Order" name="{doc_name}"'
        )

    summary = str(response).strip()

    # save to field
    frappe.db.set_value("Sales Order", doc_name, "custom_ai_summary", summary)
    frappe.db.commit()

    return summary

import frappe                          # Frappe framework
import os                              # to set env variables
from strands import Agent              # Strands Agent (brain)
from strands.models.litellm import LiteLLMModel  # LiteLLM wrapper for Groq
from strands.tools.mcp import MCPClient           # MCP tool connector
from mcp.client.streamable_http import streamablehttp_client  # HTTP transport
from functools import partial          # to pre-fill function args

# ── CONFIG ───────────────────────────────────────────────
FAC_MCP_URL = "http://localhost:8002/..."   # FAC MCP server URL
FAC_MCP_HEADERS = {"Authorization": "..."}  # auth token
GROQ_API_KEY = "gsk_..."                    # Groq API key

# ── HOOK (runs on Submit button click) ───────────────────
def generate_so_info(doc, method=None):
    frappe.enqueue(                    # dont block UI, run in background
        "...run_so_agent",
        queue="long",                  # use long queue (heavy task)
        timeout=300,                   # max 5 minutes
        doc_name=doc.name,             # pass SO name to background job
    )

# ── BACKGROUND JOB (actual AI work) ──────────────────────
def run_so_agent(doc_name):
    try:
        os.environ["GROQ_API_KEY"] = GROQ_API_KEY  # set key for LiteLLM

        transport = partial(           # create MCP connection
            streamablehttp_client, 
            FAC_MCP_URL, 
            headers=FAC_MCP_HEADERS
        )

        with MCPClient(transport) as mcp_client:      # connect to FAC MCP
            tools = mcp_client.list_tools_sync()      # get all 16 FAC tools

            model = LiteLLMModel(
                model_id="groq/llama-3.3-70b-versatile",  # Groq LLaMA model
            )

            agent = Agent(
                model=model,           # Groq as the brain
                system_prompt="You are a Frappe ERP assistant...",
                tools=tools,           # give agent FAC tools
            )

            response = agent(          # agent decides to call get_document
                f"Fetch Sales Order {doc_name} using get_document tool..."
            )

        # save AI response to field
        frappe.db.set_value("Sales Order", doc_name, "custom_ai_summary", str(response))
        frappe.db.commit()

    except Exception as e:
        frappe.log_error(frappe.get_traceback(), "SO Fetch Error")
